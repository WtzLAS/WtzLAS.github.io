#include <stdexcept>

template <typename T, long long sz=100000> class fstack {
    private:
        T *storage;
        long long ptr;

    public:
        fstack() {
            ptr = 0;
            storage = new T[sz];
        }

        ~fstack() {
            delete[] storage;
        }

        bool empty() const {
            return ptr == 0;
        }

        bool full() const {
            return ptr == sz;
        }

        long long size() const {
            return ptr;
        }

        void push(const T &elm) {
            if (full()) {
                throw new std::overflow_error("stack overflow");
            }
            storage[++ptr] = elm;
        }

        const T &top() {
            if (empty()) {
                throw new std::underflow_error("stack underflow");
            }
            return storage[ptr];
        }

        const T &peek(long long down) {
            if (ptr - down < 0) {
                throw new std::underflow_error("stack invalid argument");
            }
            return storage[ptr - down];
        }

        void pop() {
            if (empty()) {
                throw new std::underflow_error("stack underflow");
            }
            --ptr;
        }

        void pop(T &receiver) {
            if (empty()) {
                throw new std::underflow_error("stack underflow");
            }
            receiver = top();
            pop();
        }

        void clear() {
            ptr = 0;
        }
};

#include <stdexcept>

template <typename T, long long sz = 100000> class fqueue {
    private:
        T *storage;
        long long w_ptr, r_ptr, len;

    public:
        fqueue() {
            w_ptr = r_ptr = len = 0;
            storage = new T[sz];
        }

        ~fqueue() {
            delete[] storage;
        }

        bool empty() const {
            return len == 0;
        }

        bool full() const {
            return len == sz;
        }

        void push(const T &elm) {
            if (full()) {
                throw new std::overflow_error("queue overflow");
            }
            w_ptr %= sz;
            storage[w_ptr] = elm;
            ++len;
            ++w_ptr;
        }

        void pop(T &receiver) {
            if (empty()) {
                throw new std::underflow_error("queue underflow");
            }
            r_ptr %= sz;
            receiver = storage[r_ptr];
            ++r_ptr;
            --len;
        }

        void clear() {
            len = r_ptr = w_ptr = 0;
        }
};

双端队列：

#include <stdexcept>

template <typename T, long long sz = 100000> class fdeque {
    private:
        long long len;
        long long front;
        long long rear;
        T *array;

    public:
        fdeque() {
            array = new T[sz];
            len = 0;
            front = 1;
            rear = 0;
        }

        ~fdeque() {
            delete[] array;
        }

        void clear() {
            len = 0;
            front = 1;
            rear = 0;
        }

        bool empty() const {
            return len == 0;
        }

        bool full() const {
            return len == sz;
        }

        void push_front(const T &elm) {
            if (full()) {
                throw new std::overflow_error("deque full");
            }
            front = (front - 1 + sz) % sz;
            array[front] = elm;
            ++len;
        }

        const T& front() const {
            if (empty()) {
                throw new std::underflow_error("deque empty");
            }
            return array[front];
        }

        void pop_front() {
            if (empty()) {
                throw new std::underflow_error("deque empty");
            }
            front = (front + 1 + sz) % sz;
            --len;
        }

        void pop_front(T &receiver) {
            receiver = front();
            pop_front();
        }

        void push_back(const T &elm) {
            if (full()) {
                throw new std::overflow_error("deque full");
            }
            rear = (rear + 1 + sz) % sz;
            array[rear] = elm;
            ++len;
        }

        const T& back() const {
            if (empty()) {
                throw new std::underflow_error("deque empty");
            }
            return array[rear];
        }

        void pop_back() {
            if (empty()) {
                throw new std::underflow_error("deque empty");
            }
            rear = (rear - 1 + sz) % sz;
            --len;
        }

        void pop_back(T &receiver) {
            receiver = back();
            pop_back();
        }
};
